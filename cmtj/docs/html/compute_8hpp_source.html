45  const T omega = 2 * M_PI * frequency;
46  std::vector<T> &resistance = log[resTag];
47  auto it = std::find_if(log["time"].begin(), log["time"].end(),
48  [&minTime](const auto &value) { return value >= minTime; });
49  // turn into index
50  const int thresIdx = (int)(log["time"].end() - it);
51  const int cutSize = log["time"].size() - thresIdx;
52  // Rpp
53  const T RppMax = *std::max_element(resistance.begin() + thresIdx, resistance.end());
54  const T RppMin = *std::min_element(resistance.begin() + thresIdx, resistance.end());
55  const T avgR = std::accumulate(resistance.begin() + thresIdx, resistance.end(), 0.0) / cutSize;
56  const T Iampl = sqrt(power / avgR);
57  std::vector<T> voltage, current;
58  std::transform(
59  log["time"].begin() + thresIdx, log["time"].end(),
60  std::back_inserter(current),
61  [&Iampl, &omega](const T &time) { return Iampl * sin(omega * time); });
62 
63  for (unsigned int i = 0; i < cutSize; i++)
64  {
65  voltage.push_back(resistance[thresIdx + i] * current[i]);
66  }
67  const T Vmix = std::accumulate(voltage.begin(), voltage.end(), 0.0) / voltage.size();
68  std::unordered_map<std::string, T> mRes = {{"Vmix", Vmix}, {"RMax", RppMax}, {"RMin", RppMin}, {"Rpp", (RppMax - RppMin)}};
69  return mRes;
