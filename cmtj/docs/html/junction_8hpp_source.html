348  const CVector<T> llg(T time, CVector<T> m, CVector<T> bottom, CVector<T> top, T timeStep)
349  {
350  const CVector<T> heff = calculateHeff(time, timeStep, m, bottom, top);
351  const CVector<T> prod = c_cross<T>(m, heff);
352  const CVector<T> prod2 = c_cross<T>(m, prod);
353  if (this->includeSTT)
354  {
355  this->I_log = this->currentDriver.getCurrentScalarValue(time);
356  // damping-like torque factor
357  const T aJ = HBAR * this->I_log /
358  (ELECTRON_CHARGE * this->Ms * this->thickness);
359 
360  const T slonSq = pow(this->SlonczewskiSpacerLayerParameter, 2);
361  // field like
362  const T eta = (this->spinPolarisation * slonSq) / (slonSq + 1 + (slonSq - 1) * c_dot<T>(m, this->referenceLayer));
363  const T sttTerm = GYRO * aJ * eta;
364 
365  const CVector<T> prod3 = c_cross<T>(m, this->referenceLayer);
366  // first term is "damping-like torque"
367  // second term is "field-like torque"
368  CVector<T> dmdt = (prod * -GYRO) - (prod2 * GYRO * this->damping) + c_cross<T>(m, prod3) * -sttTerm + prod3 * sttTerm * this->beta;
369  return dmdt;
370  }
371  CVector<T> dmdt = (prod * -GYRO) - (prod2 * GYRO * this->damping);
372  return dmdt;
373  }

375  void rk4_step(T time, T timeStep, CVector<T> bottom, CVector<T> top)
376  {
377  CVector<T> m_t = this->mag;
378  const CVector<T> k1 = llg(time, m_t, bottom, top, timeStep) * timeStep;
379  const CVector<T> k2 = llg(time + 0.5 * timeStep, m_t + k1 * 0.5, bottom, top, timeStep) * timeStep;
380  const CVector<T> k3 = llg(time + 0.5 * timeStep, m_t + k2 * 0.5, bottom, top, timeStep) * timeStep;
381  const CVector<T> k4 = llg(time + timeStep, m_t + k3, bottom, top, timeStep) * timeStep;
382  m_t = m_t + (k1 + (k2 * 2.0) + (k3 * 2.0) + k4) / 6.0;
383  m_t.normalize();
384  this->mag = m_t;
385  }
